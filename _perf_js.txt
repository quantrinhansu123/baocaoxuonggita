
        function getKpiTargetForChannel(channelName) {
            const key = norm(channelName);
            const list = kpiTargetCache.channels || [];
            const found = list.find(function(c) { return norm(c.channel) === key; });
            return found || null;
        }

        function gradeFromScore(score) {
            const s = Math.max(0, Math.min(100, score));
            if (s > 90) return { letter: 'S', cls: 'grade-s' };
            if (s >= 70) return { letter: 'A', cls: 'grade-a' };
            if (s >= 50) return { letter: 'B', cls: 'grade-b' };
            if (s >= 30) return { letter: 'C', cls: 'grade-c' };
            return { letter: 'D', cls: 'grade-d' };
        }

        function calcPerformanceScore(stats, maxStats, kpiTarget) {
            const parts = [];
            if (kpiTarget) {
                if (kpiTarget.posts > 0) parts.push(Math.min(100, (stats.posts / kpiTarget.posts) * 100));
                if (kpiTarget.views > 0) parts.push(Math.min(100, (stats.views / kpiTarget.views) * 100));
            }
            if (parts.length) {
                return parts.reduce(function(a, b) { return a + b; }, 0) / parts.length;
            }
            const maxV = maxStats.views || 1;
            const maxE = maxStats.engagement || 1;
            return Math.min(100, (stats.views / maxV) * 70 + (maxE > 0 ? (stats.engagement / maxE) * 30 : 0));
        }

        function calcKpiPct(stats, kpiTarget) {
            if (!kpiTarget) return null;
            const parts = [];
            if (kpiTarget.posts > 0) parts.push((stats.posts / kpiTarget.posts) * 100);
            if (kpiTarget.views > 0) parts.push((stats.views / kpiTarget.views) * 100);
            if (!parts.length) return null;
            return parts.reduce(function(a, b) { return a + b; }, 0) / parts.length;
        }

        function formatGrowthPct(pct) {
            if (!Number.isFinite(pct)) return '<span class="growth-flat">—</span>';
            const cls = pct > 0 ? 'growth-up' : (pct < 0 ? 'growth-down' : 'growth-flat');
            const sign = pct > 0 ? '+' : '';
            return '<span class="' + cls + '">' + sign + formatNum(pct, 1) + '%</span>';
        }

        function groupRowsByAssignee(rows) {
            const ak = columnKeys.assignee;
            if (!ak) return [];
            const sample = rows.find(function(r) { return !isTotalRow(r) && String(r[ak] ?? '').trim(); }) || rows[0];
            const engCols = getEngagementCols(sample);
            const viewCol = columnKeys.views || findColKey(sample, ['view']);
            const groups = new Map();
            rows.filter(function(r) { return !isTotalRow(r) && String(r[ak] ?? '').trim(); }).forEach(function(row) {
                const displayName = getCellText(row, ak);
                const key = norm(displayName);
                if (!groups.has(key)) {
                    groups.set(key, { name: displayName, posts: 0, views: 0, engagement: 0 });
                }
                const g = groups.get(key);
                g.posts += 1;
                if (viewCol) g.views += parseNum(row[viewCol]);
                g.engagement += rowEngagementSum(row, engCols);
            });
            return Array.from(groups.values()).sort(function(a, b) { return b.views - a.views; });
        }

        function filterRowsByMonthOffset(monthOffset) {
            const dateCol = columnKeys.date;
            if (!dateCol) return [];
            const sel = getDefaultKpiMonthSelection();
            let y = sel.year;
            let m = sel.month + monthOffset;
            while (m < 1) { m += 12; y -= 1; }
            while (m > 12) { m -= 12; y += 1; }
            return reportData.filter(function(row) {
                if (isTotalRow(row)) return false;
                const d = parseRowDate(row[dateCol]);
                if (!d) return false;
                return d.getFullYear() === y && (d.getMonth() + 1) === m;
            });
        }

        function buildPerfStatsMap(groups, prevGroups) {
            const prevMap = new Map();
            prevGroups.forEach(function(g) { prevMap.set(norm(g.name), g); });
            return groups.map(function(g) {
                const prev = prevMap.get(norm(g.name));
                const prevViews = prev ? prev.views : 0;
                const growth = prevViews > 0 ? ((g.views - prevViews) / prevViews) * 100 : (g.views > 0 ? 100 : 0);
                return Object.assign({}, g, { prevViews: prevViews, growth: growth });
            });
        }

        function bindBestPerformer(rows) {
            const box = document.getElementById('bestPerformerContent');
            if (!box) return;
            const ak = columnKeys.assignee;
            if (!ak) {
                box.innerHTML = '<div class="best-performer-placeholder"><i class="fas fa-hourglass-half"></i> Chờ cột «Người phụ trách» từ sheet. Dashboard tự hiển thị sau khi có data.</div>';
                return;
            }
            const people = groupRowsByAssignee(rows);
            if (!people.length) {
                box.innerHTML = '<div class="best-performer-placeholder"><i class="fas fa-info-circle"></i> Chưa có dữ liệu người phụ trách trong khoảng lọc.</div>';
                return;
            }
            const maxStats = {
                views: Math.max.apply(null, people.map(function(p) { return p.views; })),
                engagement: Math.max.apply(null, people.map(function(p) { return p.engagement; }))
            };
            let best = null;
            let bestScore = -1;
            people.forEach(function(p) {
                const score = calcPerformanceScore(p, maxStats, null);
                if (score > bestScore) { bestScore = score; best = p; }
            });
            if (!best) return;
            const grade = gradeFromScore(bestScore);
            const initial = best.name.trim().charAt(0).toUpperCase() || '?';
            box.innerHTML =
                '<div class="best-performer-winner">' +
                '<div class="best-performer-avatar">' + escapeHtml(initial) + '</div>' +
                '<div><div class="best-performer-name">' + escapeHtml(best.name) + '</div>' +
                '<div class="best-performer-meta">Xếp loại <span class="grade-badge ' + grade.cls + '">' + grade.letter + '</span> · Điểm ' + formatNum(bestScore, 1) + '</div></div></div>' +
                '<div class="best-performer-stats">' +
                '<div class="best-performer-stat"><strong>' + formatNum(best.posts) + '</strong><span>Bài đăng</span></div>' +
                '<div class="best-performer-stat"><strong>' + formatNum(best.views) + '</strong><span>Lượt xem</span></div>' +
                '<div class="best-performer-stat"><strong>' + formatNum(best.engagement) + '</strong><span>Tương tác</span></div>' +
                '</div>';
        }

        function bindPerfChannelTable(rows) {
            const tbody = document.getElementById('perfChannelBody');
            if (!tbody) return;
            const grouped = groupRowsByChannel(rows);
            if (!grouped.length) {
                tbody.innerHTML = '<tr><td colspan="7" style="text-align:center;color:#999;padding:16px;">Chưa có dữ liệu kênh</td></tr>';
                return;
            }
            const maxStats = {
                views: Math.max.apply(null, grouped.map(function(g) { return g.views; })),
                engagement: Math.max.apply(null, grouped.map(function(g) { return g.engagement; }))
            };
            tbody.innerHTML = grouped.map(function(g, i) {
                const kpi = getKpiTargetForChannel(g.name);
                const score = calcPerformanceScore(g, maxStats, kpi);
                const grade = gradeFromScore(score);
                return '<tr>' +
                    '<td>' + (i + 1) + '</td>' +
                    '<td>' + escapeHtml(g.name) + '</td>' +
                    '<td class="num">' + formatNum(g.posts) + '</td>' +
                    '<td class="num">' + formatNum(g.views) + '</td>' +
                    '<td class="num">' + formatNum(g.engagement) + '</td>' +
                    '<td class="num">' + formatNum(score, 1) + '</td>' +
                    '<td><span class="grade-badge ' + grade.cls + '">' + grade.letter + '</span></td>' +
                    '</tr>';
            }).join('');
        }

        function bindPerfStaffTable(rows) {
            const tbody = document.getElementById('perfStaffBody');
            const sub = document.getElementById('perfStaffSubtitle');
            if (!tbody) return;
            const ak = columnKeys.assignee;
            const useAssignee = !!ak;
            if (sub) sub.textContent = useAssignee ? 'Theo người phụ trách' : 'Tạm theo kênh';
            const currentGroups = useAssignee ? groupRowsByAssignee(rows) : groupRowsByChannel(rows);
            const prevRows = filterRowsByMonthOffset(-1);
            const prevGroups = useAssignee ? groupRowsByAssignee(prevRows) : groupRowsByChannel(prevRows);
            const stats = buildPerfStatsMap(currentGroups, prevGroups);
            if (!stats.length) {
                tbody.innerHTML = '<tr><td colspan="9" style="text-align:center;color:#999;padding:16px;">Chưa có dữ liệu</td></tr>';
                return;
            }
            const maxStats = {
                views: Math.max.apply(null, stats.map(function(s) { return s.views; })),
                engagement: Math.max.apply(null, stats.map(function(s) { return s.engagement; }))
            };
            tbody.innerHTML = stats.map(function(s, i) {
                const kpi = useAssignee ? null : getKpiTargetForChannel(s.name);
                const score = calcPerformanceScore(s, maxStats, kpi);
                const grade = gradeFromScore(score);
                const kpiPct = calcKpiPct(s, kpi);
                const detail = 'Bài: ' + formatNum(s.posts) + ' · View: ' + formatNum(s.views) + ' · TT: ' + formatNum(s.engagement) +
                    (kpiPct != null ? ' · KPI: ' + formatNum(kpiPct, 1) + '%' : '');
                return '<tr>' +
                    '<td>' + (i + 1) + '</td>' +
                    '<td>' + escapeHtml(s.name) + '</td>' +
                    '<td class="num">' + formatNum(s.posts) + '</td>' +
                    '<td class="num">' + formatNum(s.views) + '</td>' +
                    '<td class="num">' + formatGrowthPct(s.growth) + '</td>' +
                    '<td class="num">' + (kpiPct != null ? formatNum(kpiPct, 1) + '%' : '—') + '</td>' +
                    '<td class="num">' + formatNum(score, 1) + '</td>' +
                    '<td><span class="grade-badge ' + grade.cls + '">' + grade.letter + '</span></td>' +
                    '<td><button type="button" class="perf-detail-btn" title="' + escapeHtml(detail) + '">Xem</button></td>' +
                    '</tr>';
            }).join('');
            tbody.querySelectorAll('.perf-detail-btn').forEach(function(btn) {
                btn.addEventListener('click', function() {
                    showDashToast(btn.getAttribute('title') || '', 'success');
                });
            });
        }

        function bindPerformanceEvaluation(rows) {
            bindBestPerformer(rows);
            bindPerfChannelTable(rows);
            bindPerfStaffTable(rows);
        }
